


import time
servidores=[]
creditos=[]


def menu():
    print("Olá! Seja Bem Vindo!, digite o numero correspondente a função desejada: ")
    time.sleep(2)
    print("[1] - Novo Cadastro [2] - Adicionar Saldo [3] - Consultar Saldo")
    print("[4] - Acessar Refeitório  [5] - Listar Servidores")
    entrada = int(input("Digite o número aqui: "))
    return entrada
    


def cadastrar_servidor():
    print("Cadastrar Servidor")
    nome = input("Nome: ")
    servidores.append(nome)
    creditos.append(0)
    print("Aguarde um momento...")
    time.sleep(2)
    print(f"Servidor : {nome} cadastrado com sucesso!")
    time.sleep(2)
    
def listar_servidores():
    print("Listar Servidores")
    for i in range(len(servidores)):
        print(f"Nome: ['  {servidores[i]}  '],  CÓDIGO: [  {i}  ]")
            
def adicionar_credito():
    print("Edicionar Crédito")
    codigo = int(input("Qual o código do servidor?: "))
    novo = float(input("Qual o saldo acrescentado?: "))
    creditos[codigo] = creditos[codigo] + novo
    print("Aguarde um momento...")
    time.sleep(2)
    print(f"O saldo de [  R${novo}  ]  para, [ {servidores[codigo]} ] adicionado com sucesso!!")
    time.sleep(2)
    
def consultar_saldo():
    print("Consultar Saldo disponível")
    codigo = int(input("Qual o código do servidor?: "))
    if creditos[codigo] == 0:
        print("Aguarde um momento...")
        time.sleep(2)
        print(f"Nome: ['  {servidores[codigo]}  '], CRÉDITO: [  R${creditos[codigo]}  ] SALDO INSUFICIENTE")
        time.sleep(2)
    else:
        print("Aguarde um momento...")
        time.sleep(2)
        print(f"Nome: ['  {servidores[codigo]}  '], CRÉDITO: [  R${creditos[codigo]}  ] SALDO DISPONÍVEL")
        time.sleep(2)
      
def acessar_refeitorio():
    print("Acessar Refeitório")
    codigo = int(input("Qual o código do servidor?: "))
    if creditos[codigo] < 2:
        print("Aguarde um momento...")
        time.sleep(2)
        print(f"Nome: ['  {servidores[codigo]}  '], CRÉDITO ATUAL: [  R${creditos[codigo]}  ] ACESSO NEGADO")
        time.sleep(2)
    else:
        print("Aguarde um momento...")
        time.sleep(2)
        print(f"Nome: ['  {servidores[codigo]}  ']")
        print("ACESSO LIBERADO!!")
        creditos[codigo] = creditos[codigo] -2
        time.sleep(2)
    
escolha_menu =-1
while escolha_menu != 0:
    escolha_menu = menu()
    if escolha_menu == 1:
        cadastrar_servidor()
    elif escolha_menu == 2:
        adicionar_credito()
    elif escolha_menu == 3:
        consultar_saldo()
    elif escolha_menu == 4:
        acessar_refeitorio()
    elif escolha_menu == 5:
        
        listar_servidores()
    else:
        print("Obrigado pela visita")
        time.sleep(2)
